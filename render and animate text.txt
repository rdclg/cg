import pygame
pygame.init()
# Create window
window = pygame.display.set_mode((500, 300))
pygame.display.set_caption("Text Rendering Example")
# Font and Text
font = pygame.font.Font(None, 50)        # Default font, size 50
text = font.render("Hello Pygame!", True, (255, 0, 0))  # Red text
x = 0   # start position
running = True
while running:
    window.fill((230, 240, 255))         # Background color
    for event in pygame.event.get():
        if event.type == pygame.QUIT:
            running = False
    window.blit(text, (x, 120))          # Draw text at position (x,120)
    x += 0.01                               # Move text slowly to right
    if x > 500:                          # If text goes out, reset
        x = 0
    pygame.display.update()              # Update screen
pygame.quit()

import pygame
#Initilize Pygame
pygame.init()
# Make window
win = pygame.display.set_mode((400, 400))
x,y = 50,150 # ball start position
running = True
while running:
    for event in pygame.event.get():
        if event.type == pygame.QUIT:
            running = False
    win.fill((255, 255, 255))       # clear window (white)
    pygame.draw.circle(win, (255, 0, 0), (x, y), 20)  # draw red ball
    x += 2                             # move ball to right
    if x > 400:                        # reset if out of window
        x = 0
    pygame.display.update()            # show new frame
    pygame.time.delay(20)              # small pause for speed control
pygame.quit()

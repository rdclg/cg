import pygame
pygame.init()
# Window setup
window = pygame.display.set_mode((500, 400))
pygame.display.set_caption("Single Click Circle Move")
# Circle start position
x, y = 250, 200
running = True
while running:
    window.fill((255, 255, 255))   # white background
    for event in pygame.event.get():
        if event.type == pygame.QUIT:
            running = False
        # When mouse is clicked once
        if event.type == pygame.MOUSEBUTTONDOWN:
            x, y = event.pos  # Change position to click point
    # Draw circle at new (x, y)
    pygame.draw.circle(window, (255, 0, 0), (x, y), 30)
    pygame.display.update()
pygame.quit()
